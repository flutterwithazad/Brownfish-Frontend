const Button = ({ 
  children, 
  variant = 'primary', 
  onClick, 
  className = '',
  disabled = false,
  type = 'button'
}: { 
  children: React.ReactNode; 
  variant?: 'primary' | 'secondary' | 'outline'; 
  onClick?: () => void;
  className?: string;
  disabled?: boolean;
  type?: 'button' | 'submit';
}) => {
  const baseStyles = "inline-flex items-center justify-center px-6 py-3 text-base font-medium transition-all duration-200 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-amber-500 disabled:opacity-50 disabled:cursor-not-allowed";
  
  const variants = {
    primary: `${THEME.colors.accentBg} text-white ${THEME.colors.accentHover} shadow-lg shadow-amber-900/20`,
    secondary: "bg-white text-neutral-900 hover:bg-neutral-100",
    outline: `border border-neutral-700 text-neutral-300 hover:border-amber-600 hover:text-white bg-transparent`
  };

  return (
    <button type={type} onClick={onClick} disabled={disabled} className={`${baseStyles} ${variants[variant]} ${className}`}>
      {children}
    </button>
  );
};

const ContactPage = ({ user }: { user: any }) => {
  const [formData, setFormData] = useState({
    name: '',
    email: '',
    type: 'Web Application Development',
    message: ''
  });
  const [status, setStatus] = useState<'idle' | 'submitting' | 'success' | 'error'>('idle');

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!user) return; // Guard against unauthed submission

    setStatus('submitting');

    try {
      // Save to Firestore (Simulating "Storage")
      // Strict path: /artifacts/{appId}/public/data/enquiries
      await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'enquiries'), {
        ...formData,
        timestamp: serverTimestamp(),
        userId: user.uid
      });

      setStatus('success');
      setFormData({ name: '', email: '', type: 'Web Application Development', message: '' });
      
      // Reset success message after 5 seconds
      setTimeout(() => setStatus('idle'), 5000);

    } catch (error) {
      console.error("Error submitting form: ", error);
      setStatus('error');
    }
  };

  const handleChange = (e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement | HTMLSelectElement>) => {
    setFormData(prev => ({
      ...prev,
      [e.target.id]: e.target.value
    }));
  };

  return (
    <div className="animate-fade-in py-20 md:py-32">
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
        <SectionTitle subtitle="Start a Conversation">Let's Build Something Scalable.</SectionTitle>
        
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-16">
          <div>
            <p className="text-xl text-neutral-300 mb-8">
              Tell us about your project constraints, timeline, and goals. We'll give you a straight answer on feasibility and cost.
            </p>
            
            <div className="space-y-6 mb-12">
              <a href="mailto:hello@brownfish.tech" className="flex items-center gap-4 text-neutral-400 hover:text-white transition-colors">
                <Mail className="w-5 h-5" />
                <span>hello@brownfish.tech</span>
              </a>
              <div className="flex items-center gap-4 text-neutral-400">
                <Github className="w-5 h-5" />
                <span>github.com/brownfish-tech</span>
              </div>
              <div className="flex items-center gap-4 text-neutral-400">
                <Linkedin className="w-5 h-5" />
                <span>linkedin.com/company/brown-fish</span>
              </div>
            </div>

            <div className="p-6 bg-neutral-900 border border-neutral-800">
              <h4 className="font-bold text-white mb-2">Not ready to quote?</h4>
              <p className="text-sm text-neutral-400">
                Book a 15-minute discovery call to see if we're a good fit. No sales pressure, just engineering chat.
              </p>
            </div>
          </div>

          <form className="space-y-6 relative" onSubmit={handleSubmit}>
             {status === 'success' && (
              <div className="absolute inset-0 bg-neutral-950/90 flex flex-col items-center justify-center z-10 border border-green-900 rounded">
                <div className="w-16 h-16 bg-green-900/30 rounded-full flex items-center justify-center mb-4">
                  <Check className="w-8 h-8 text-green-500" />
                </div>
                <h3 className="text-xl font-bold text-white mb-2">Message Received</h3>
                <p className="text-neutral-400 text-center max-w-xs">
                  Thanks for reaching out. Our engineering team will review your enquiry and respond within 24 hours.
                </p>
                <button 
                  type="button" 
                  onClick={() => setStatus('idle')}
                  className="mt-6 text-sm text-neutral-500 hover:text-white underline"
                >
                  Send another message
                </button>
              </div>
            )}

            <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
              <div className="space-y-2">
                <label htmlFor="name" className="text-sm font-medium text-neutral-400">Name</label>
                <input 
                  type="text" 
                  id="name" 
                  required
                  value={formData.name}
                  onChange={handleChange}
                  className="w-full bg-neutral-950 border border-neutral-800 rounded-none px-4 py-3 text-white focus:border-amber-600 focus:ring-1 focus:ring-amber-600 outline-none transition-all" 
                  placeholder="Jane Doe" 
                />
              </div>
              <div className="space-y-2">
                <label htmlFor="email" className="text-sm font-medium text-neutral-400">Work Email</label>
                <input 
                  type="email" 
                  id="email" 
                  required
                  value={formData.email}
                  onChange={handleChange}
                  className="w-full bg-neutral-950 border border-neutral-800 rounded-none px-4 py-3 text-white focus:border-amber-600 focus:ring-1 focus:ring-amber-600 outline-none transition-all" 
                  placeholder="jane@company.com" 
                />
              </div>
            </div>
            
            <div className="space-y-2">
              <label htmlFor="type" className="text-sm font-medium text-neutral-400">Project Type</label>
              <select 
                id="type" 
                value={formData.type}
                onChange={handleChange}
                className="w-full bg-neutral-950 border border-neutral-800 rounded-none px-4 py-3 text-white focus:border-amber-600 focus:ring-1 focus:ring-amber-600 outline-none transition-all appearance-none"
              >
                <option>Web Application Development</option>
                <option>Mobile App (iOS/Android)</option>
                <option>Backend/API Development</option>
                <option>MVP Build</option>
                <option>Legacy Code Audit</option>
              </select>
            </div>

            <div className="space-y-2">
              <label htmlFor="message" className="text-sm font-medium text-neutral-400">Project Details</label>
              <textarea 
                id="message" 
                rows={5} 
                required
                value={formData.message}
                onChange={handleChange}
                className="w-full bg-neutral-950 border border-neutral-800 rounded-none px-4 py-3 text-white focus:border-amber-600 focus:ring-1 focus:ring-amber-600 outline-none transition-all" 
                placeholder="Briefly describe what you're building..."
              ></textarea>
            </div>

            <Button 
              type="submit" 
              className="w-full md:w-auto min-w-[160px]"
              disabled={status === 'submitting'}
            >
              {status === 'submitting' ? (
                <><Loader2 className="w-4 h-4 mr-2 animate-spin" /> Sending...</>
              ) : (
                'Send Inquiry'
              )}
            </Button>
            
            {status === 'error' && (
              <p className="text-red-500 text-sm mt-2">
                Something went wrong. Please try again or email us directly.
              </p>
            )}
          </form>
        </div>
      </div>
    </div>
  );
};

        <p className="mt-2 md:mt-0 font-mono">Designed & Engineered in New Delhi.</p>
      </div>
    </div>
  </footer>
);

export default function App() {
  const [activePage, setActivePage] = useState<Page>('home');
  const [isMenuOpen, setIsMenuOpen] = useState(false);
  const [scrolled, setScrolled] = useState(false);
  const [user, setUser] = useState<any>(null);

  // --- Auth Initialization ---
  useEffect(() => {
    const initAuth = async () => {
      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        await signInWithCustomToken(auth, __initial_auth_token);
      } else {
        await signInAnonymously(auth);
      }
    };
    initAuth();
    
    const unsubscribe = onAuthStateChanged(auth, setUser);
    return () => unsubscribe();
  }, []);

  // Handle scroll effect for navbar
  useEffect(() => {
    const handleScroll = () => setScrolled(window.scrollY > 20);
    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  // Scroll to top on page change
  useEffect(() => {
    window.scrollTo(0, 0);
    setIsMenuOpen(false);
  }, [activePage]);

  const navLinks: { label: string; value: Page }[] = [
    { label: 'Home', value: 'home' },
    { label: 'Services', value: 'services' },
    { label: 'Projects', value: 'projects' },
    { label: 'About', value: 'about' },
  ];

  return (
    <div className="min-h-screen bg-neutral-950 text-neutral-200 font-sans selection:bg-amber-900 selection:text-white">
      {/* Navigation */}
      <nav className={`fixed w-full z-50 transition-all duration-300 border-b ${scrolled ? 'bg-neutral-950/90 backdrop-blur-md border-neutral-800 py-4' : 'bg-transparent border-transparent py-6'}`}>
        <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
          <div className="flex justify-between items-center">
            {/* Logo */}
            <div 
              className="flex items-center gap-3 cursor-pointer group" 
              onClick={() => setActivePage('home')}
            >
              <div className={`w-10 h-10 rounded bg-amber-700 flex items-center justify-center transition-transform group-hover:scale-105 shadow-lg shadow-amber-900/20`}>
                 {/* Replaced Image with Icon for reliability in preview */}
                 <Cpu className="text-white w-6 h-6" />
              </div>
              <div className="flex flex-col">
                <span className="text-lg font-bold text-white leading-none tracking-tight">BROWN FISH</span>
                <span className="text-[10px] uppercase tracking-[0.2em] text-neutral-500 font-mono">Technologies</span>
              </div>
            </div>

            {/* Desktop Menu */}
            <div className="hidden md:flex items-center gap-8">
              {navLinks.map((link) => (
                <button
                  key={link.value}
                  onClick={() => setActivePage(link.value)}
                  className={`text-sm font-medium transition-colors hover:text-white ${activePage === link.value ? 'text-white' : 'text-neutral-400'}`}
                >
                  {link.label}
                </button>
              ))}
              <Button onClick={() => setActivePage('contact')} className="!py-2 !px-4 text-sm">
                Get in Touch
              </Button>
            </div>

            {/* Mobile Menu Toggle */}
            <button 
              className="md:hidden text-white p-2"
              onClick={() => setIsMenuOpen(!isMenuOpen)}
            >
              {isMenuOpen ? <X /> : <Menu />}
            </button>
          </div>
        </div>

        {/* Mobile Menu Overlay */}
        {isMenuOpen && (
          <div className="md:hidden absolute top-full left-0 w-full bg-neutral-900 border-b border-neutral-800 py-4 px-4 flex flex-col gap-4 shadow-2xl">
            {navLinks.map((link) => (
              <button
                key={link.value}
                onClick={() => setActivePage(link.value)}
                className={`text-left text-lg font-medium py-2 px-4 rounded hover:bg-neutral-800 ${activePage === link.value ? 'text-amber-500' : 'text-neutral-300'}`}
              >
                {link.label}
              </button>
            ))}
            <button 
              onClick={() => setActivePage('contact')}
              className="text-left text-lg font-medium py-2 px-4 rounded text-amber-500 hover:bg-neutral-800"
            >
              Contact
            </button>
          </div>
        )}
      </nav>

      {/* Main Content */}
      <main className="min-h-screen pt-20">
        {activePage === 'home' && <HomePage navigateTo={setActivePage} />}
        {activePage === 'services' && <ServicesPage />}
        {activePage === 'projects' && <ProjectsPage />}
        {activePage === 'about' && <AboutPage />}
        {activePage === 'contact' && <ContactPage user={user} />}
      </main>

      <Footer navigateTo={setActivePage} />
    </div>
  );
}
